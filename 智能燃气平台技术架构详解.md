```

                                                          │ JDBC

                                                  └──────────────┘
```

#### （2）Axios拦截器封装

**文件位置：** `src/main.js:110-160`

```javascript
// 创建Axios实例
const httpInstance = axios.create({
    baseURL: baseUrl,
    timeout: 30000,
    headers: { 'Content-Type': 'application/json' }
})

// 请求拦截器：自动添加Token
httpInstance.interceptors.request.use(
    config => {
        const token = localStorage.getItem("token")
        if (token) {
            config.headers['token'] = token  // Sa-Token要求
        }
        return config
    },
    error => {
        return Promise.reject(error)
    }
)

// 响应拦截器：统一错误处理
httpInstance.interceptors.response.use(
    response => response.data,
    error => {
        if (error.response) {
            const status = error.response.status
            if (status === 401) {
                ElMessage.error({
                    message: "未登录或登录已过期",
                    duration: 1200
                })
                router.push({ name: 'Login' })
            } else {
                ElMessage.error({
                    message: error.response.data.error || "请求失败",
                    duration: 1200
                })
            }
        }
        return Promise.reject(error)
    }
)

// 封装全局HTTP方法
app.config.globalProperties.$http = function(url, method, data, async = true, fun) {
    return new Promise((resolve, reject) => {
        const config = {
            url: url,
            method: method
        }
        if (method.toUpperCase() === 'GET') {
            config.params = data
        } else {
            config.data = data
        }
        httpInstance(config).then(response => {
            if (fun) {
                fun(response)
            }
            resolve(response)
        }).catch(error => {
            reject(error)
        })
    })
}
```

**技术知识点：**

- **拦截器模式**：AOP（面向切面编程）思想，在不修改业务代码的情况下增强功能
- **Token认证**：采用Token-based认证，无状态、可扩展
- **统一错误处理**：避免在每个请求中重复编写错误处理代码
- **Promise封装**：支持async/await语法，提高代码可读性

#### （3）路由守卫与权限控制

**文件位置：** `src/router/index.js:92-122`

```javascript
import { createRouter, createWebHashHistory } from 'vue-router'

const routes = [
    {
        path: '/login',
        name: 'Login',
        component: () => import('../views/login.vue')
    },
    {
        path: '/',
        name: 'Main',
        component: () => import("../views/main.vue"),
        children: [
            {
                path: '/home',
                name: 'Home',
                component: () => import("../views/home.vue"),
                meta: { title: '首页' }
            },
            {
                path: "/role",
                name: "Role",
                component: () => import("../views/role.vue"),
                meta: {
                    title: "角色管理",
                    isTab: true,
                    permission: "MIS_ROLE:SELECT"  // 权限标识
                }
            },
            // ... 其他路由
        ]
    }
]

const router = createRouter({
    history: createWebHashHistory(),
    routes
})

// 全局前置守卫
router.beforeEach((to, from, next) => {
    // 1. 登录检查
    if (to.name != "Login") {
        let permissions = localStorage.getItem("permissions")
        let token = localStorage.getItem("token")
        
        if (!permissions || !token) {
            next({ name: 'Login' })
            return
        }
        
        // 2. 权限检查
        if (to.meta.permission) {
            let permissionList = []
            try {
                permissionList = JSON.parse(permissions)
            } catch (e) {
                console.error('Failed to parse permissions:', e)
                next({ name: 'Login' })
                return
            }
            
            // ROOT权限或具体权限才能访问
            if (!permissionList.includes('ROOT') && 
                !permissionList.includes(to.meta.permission)) {
                next({ name: 'Home' })
                return
            }
        }
    }
    next()
})

export default router
```

**技术知识点：**

- **导航守卫**：`beforeEach`全局前置守卫，在路由跳转前执行
- **路由元信息**：`meta`字段存储路由附加信息（如权限要求、标题）
- **权限分级**：ROOT超级管理员 > 普通权限（如MIS_ROLE:SELECT）
- **路由懒加载**：`() => import()`动态导入，按需加载组件

### 3.2 核心实现机制

#### （1）Controller层：RESTful API设计

**文件位置：** `com/example/gas/api/controller/MisRoleController.java`

```java
@RestController
@RequestMapping("/role")
public class MisRoleController {
    
    @Resource
    private MisRoleService misRoleService;
    
    /**
     * 查询角色列表（分页）
     */
    @PostMapping("/list")
    public R list(@RequestBody Map<String, Object> params) {
        // 权限检查
        if (!StpUtil.hasRole("ROOT") && 
            !StpUtil.hasPermission("MIS_ROLE:SELECT")) {
            return R.error().put("msg", "无权限访问");
        }
        
        // 分页处理
        int page = params.get("page") != null ? 
                   Integer.parseInt(params.get("page").toString()) : 1;
        int length = params.get("length") != null ? 
                     Integer.parseInt(params.get("length").toString()) : 10;
        int start = (page - 1) * length;
        params.put("start", start);
        params.put("length", length);
        
        // 调用Service
        ArrayList<HashMap> roleList = misRoleService.searchRoleListWithCount(params);
        int totalCount = misRoleService.searchRoleCount(params);
        
        return R.ok().put("list", roleList).put("total", totalCount);
    }
    
    /**
     * 查询角色详情
     */
    @GetMapping("/info/{id}")
    public R info(@PathVariable("id") Integer id) {
        if (!StpUtil.hasRole("ROOT") && 
            !StpUtil.hasPermission("MIS_ROLE:SELECT")) {
            return R.error().put("msg", "无权限访问");
        }
        MisRoleEntity role = misRoleService.searchRoleById(id);
        return R.ok().put("role", role);
    }
    
    /**
     * 新增角色
     */
    @PostMapping("/save")
    public R save(@RequestBody MisRoleEntity role) {
        if (!StpUtil.hasRole("ROOT") && 
            !StpUtil.hasPermission("MIS_ROLE:INSERT")) {
            return R.error().put("msg", "无权限访问");
        }
        Integer nextId = misRoleService.getNextId();
        role.setId(nextId);
        int result = misRoleService.insertRole(role);
        if (result > 0) {
            return R.ok();
        } else {
            return R.error();
        }
    }
    
    /**
     * 更新角色
     */
    @PostMapping("/update")
    public R update(@RequestBody MisRoleEntity role) {
        if (!StpUtil.hasRole("ROOT") && 
            !StpUtil.hasPermission("MIS_ROLE:UPDATE")) {
            return R.error().put("msg", "无权限访问");
        }
        // 禁止修改超级管理员
        if (role.getId() != null && role.getId() == 0) {
            return R.error().put("msg", "超级管理员不可修改");
        }
        int result = misRoleService.updateRole(role);
        if (result > 0) {
            return R.ok();
        } else {
            return R.error();
        }
    }
    
    /**
     * 删除角色
     */
    @PostMapping("/delete")
    public R delete(@RequestBody Map<String, Object> params) {
        if (!StpUtil.hasRole("ROOT") && 
            !StpUtil.hasPermission("MIS_ROLE:DELETE")) {
            return R.error().put("msg", "无权限访问");
        }
        Integer id = (Integer) params.get("id");
        // 禁止删除超级管理员
        if (id != null && id == 0) {
            return R.error().put("msg", "超级管理员不可删除");
        }
        int result = misRoleService.deleteRole(id);
        if (result > 0) {
            return R.ok();
        } else {
            return R.error();
        }
    }
    
    /**
     * 查询角色的权限列表
     */
    @GetMapping("/permissions/{roleId}")
    public R permissions(@PathVariable("roleId") Integer roleId) {
        if (!StpUtil.hasRole("ROOT") && 
            !StpUtil.hasPermission("MIS_ROLE:SELECT")) {
            return R.error().put("msg", "无权限访问");
        }
        List<String> permissions = misRoleService.searchRolePermissions(roleId);
        return R.ok().put("permissions", permissions);
    }
    
    /**
     * 保存角色权限
     */
    @PostMapping("/save-permissions")
    public R savePermissions(@RequestBody Map<String, Object> params) {
        if (!StpUtil.hasRole("ROOT") && 
            !StpUtil.hasPermission("MIS_ROLE:UPDATE")) {
            return R.error().put("msg", "无权限访问");
        }
        Integer roleId = (Integer) params.get("roleId");
        // 禁止修改超级管理员权限
        if (roleId != null && roleId == 0) {
            return R.error().put("msg", "超级管理员拥有所有权限，无需分配");
        }
        List<String> permissions = (List<String>) params.get("permissions");
        // 禁止为角色赋予ROOT权限
        if (permissions != null && permissions.contains("ROOT")) {
            return R.error().put("msg", "不能为角色赋予ROOT权限");
        }
        int result = misRoleService.saveRolePermission(roleId, permissions);
        if (result > 0) {
            return R.ok();
        } else {
            return R.error();
        }
    }
}
```

**技术知识点：**

- **@RestController**：组合注解 = @Controller + @ResponseBody，方法返回值自动序列化为JSON
- **@RequestMapping**：定义URL映射，支持RESTful风格
- **@Resource**：依赖注入，按名称装配（推荐使用@Autowired按类型装配）
- **RESTful设计原则**：
  - POST /role/save - 新增
  - POST /role/update - 更新
  - POST /role/delete - 删除
  - GET /role/info/{id} - 查询
  - POST /role/list - 列表查询（使用POST因为参数可能较多）
- **权限控制**：每个接口都进行权限验证，确保安全性

```java
@Service
public class MisRoleServiceImpl implements MisRoleService {
    
    @Resource
    private MisRoleDao misRoleDao;
    
    @Override
    public List<MisRoleEntity> searchRoleList(Map param) {
        return misRoleDao.searchRoleList(param);
    }
    
    @Override
    public MisRoleEntity searchRoleById(Integer id) {
        return misRoleDao.searchRoleById(id);
    }
    
    @Override
    public int insertRole(MisRoleEntity role) {
        return misRoleDao.insertRole(role);
    }
    
    @Override
    public int updateRole(MisRoleEntity role) {
        return misRoleDao.updateRole(role);
    }
    
    @Override
    public int deleteRole(Integer id) {
        return misRoleDao.deleteRole(id);
    }
    
    @Override
    public int batchDeleteRole(List<Integer> ids) {
        return misRoleDao.batchDeleteRole(ids);
    }
    
    @Override
    public List<String> searchRolePermissions(Integer roleId) {
        return misRoleDao.searchRolePermissions(roleId);
    }
    
    @Override
    public int saveRolePermission(Integer roleId, List<String> permissions) {
        // 1. 先删除原有权限
        misRoleDao.deleteRolePermission(roleId);
        
        // 2. 获取当前最大权限ID
        Integer maxPermissionId = misRoleDao.getMaxPermissionId();
        int currentId = maxPermissionId == null ? 1 : maxPermissionId + 1;
        
        // 3. 批量插入新权限
        for (String permission : permissions) {
            Map<String, Object> param = new HashMap<>();
```

```java
    @ResponseBody
    @ResponseStatus(HttpStatus.INTERNAL_SERVER_ERROR)
    @ExceptionHandler(Exception.class)
    public String exceptionHandler(Exception e) {
        JSONObject json = new JSONObject();
        
        if (e instanceof HttpMessageNotReadableException) {
            HttpMessageNotReadableException exception = (HttpMessageNotReadableException) e;
            log.error("error", exception);
            json.set("error", "请求未提交数据或者数据有误");
        } 
        else if (e instanceof MissingServletRequestPartException) {
            MissingServletRequestPartException exception = (MissingServletRequestPartException) e;
            log.error("error", exception);
            json.set("error", "请求提交数据错误");
        } 
        else if (e instanceof HttpRequestMethodNotSupportedException) {
            HttpRequestMethodNotSupportedException exception = (HttpRequestMethodNotSupportedException) e;
            log.error("error", exception);
            json.set("error", "HTTP请求方法类型错误");
        }
        // 处理后端验证失败产生的异常
        else if (e instanceof MethodArgumentNotValidException) {
            MethodArgumentNotValidException exception = (MethodArgumentNotValidException) e;
            json.set("error", exception.getBindingResult().getFieldError().getDefaultMessage());
        }
        // 处理业务异常
        else if (e instanceof HospitalException) {
            log.error("执行异常", e);
            HospitalException exception = (HospitalException) e;
            json.set("error", exception.getMsg());
        } 
        else if (e instanceof PayException) {
            PayException exception = (PayException) e;
            log.error("微信支付异常", exception);
            json.set("error", "微信支付异常");
        }
        // 处理其余的异常
        else {
            log.error("执行异常", e);
            json.set("error", "执行异常");
        }
        
        return json.toString();
    }
    
    @ResponseBody
    @ResponseStatus(HttpStatus.UNAUTHORIZED)
    @ExceptionHandler(NotLoginException.class)
    public String unLoginHandler(Exception e) {
        JSONObject json = new JSONObject();
        json.set("error", e.getMessage());
        return json.toString();
    }
}
```

## 四、认证授权机制详解

### 4.1 Sa-Token权限框架

**核心概念：**

- **Token**：用户登录后生成的令牌，存储在Redis中
- **权限（Permission）**：细粒度的操作权限，如`MIS_ROLE:SELECT`
- **角色（Role）**：权限的集合，如`ROOT`、`MANAGER`

**RBAC模型：**

```
用户（User） ←→ 用户角色关联表（MIS_USER_ROLE） ←→ 角色（Role）
                                                        ↓
                                            角色权限关联表（MIS_ROLE_PERMISSION）
                                                        ↓
                                                权限（Permission）
```

```java
@Component
public class StpInterfaceImpl implements StpInterface {
    
    @Autowired
    private MisUserDao userDao;
    
    /**
     * 返回一个用户所拥有的权限集合
     */
    @Override
    public List<String> getPermissionList(Object loginId, String loginKey) {
        int userId = Integer.parseInt(loginId.toString());
        System.out.println("LoginId: " + loginId);
        
        // 查询用户的权限列表
        ArrayList<String> list = userDao.searchUserPermissions(userId);
        
        // 超级管理员添加ROOT权限
        if (userId == 0) {
            list.add("ROOT");
        }
        
        System.out.println("Permissions: " + list);
        return list;
    }
    
    /**
     * 返回一个用户所拥有的角色标识集合
     */
    @Override
    public List<String> getRoleList(Object loginId, String loginKey) {
        int userId = Integer.parseInt(loginId.toString());
        
        // 查询用户的角色ID列表
        List<Integer> roleIds = userDao.searchUserRoles(userId);
        
        // 将角色ID转换为角色名称
        List<String> roles = new ArrayList<>();
        for (Integer roleId : roleIds) {
            if (roleId == 0) {
                roles.add("ROOT");
            } else if (roleId == 1) {
                roles.add("MANAGER");
            } else if (roleId == 2) {
                roles.add("TECH");
            } else if (roleId == 3) {
                roles.add("MARKET");
            } else if (roleId == 4) {
                roles.add("FINANCE");
            }
        }
        
        return roles;
    }
}
```

### 5.3 跨域处理

#### 什么是跨域？

当一个请求URL的**协议、域名、端口**三者之间任意一个与当前页面URL不同即为跨域。

**示例：**

| 当前页面URL | 请求URL | 是否跨域 | 原因 |
|------------|---------|---------|------|
| http://localhost:5173 | http://localhost:8092 | 是 | 端口不同 |
| http://localhost:5173 | http://127.0.0.1:8092 | 是 | 域名不同 |
| http://localhost:5173 | https://localhost:8092 | 是 | 协议不同 |

#### 跨域解决方案

**后端CORS配置：**

```java
@Override
public void addCorsMappings(CorsRegistry registry) {
    registry.addMapping("/**")
            .allowedOriginPatterns("*")  // 允许所有源
            .allowCredentials(true)      // 允许携带Cookie
            .allowedMethods("GET", "POST", "DELETE", "PUT", "PATCH")
            .maxAge(3600);               // 预检请求缓存时间
}
```

**技术知识点：**

- **CORS**：Cross-Origin Resource Sharing，跨域资源共享
- **预检请求**：浏览器在发送实际请求前，先发送OPTIONS请求询问服务器是否允许
- **allowCredentials**：允许前端携带认证信息（如Cookie、Token）

```javascript
dataFormSubmit() {
    this.$refs.dataForm.validate((valid) => {
        if (valid) {
            let that = this
            that.loading = true
            let url = this.dataForm.id ? '/role/update' : '/role/save'
            
            // 1. 先保存角色基本信息
            that.$http(url, 'POST', this.dataForm, true, function(resp) {
                if (resp.code === 0 || resp.code === 200) {
                    const roleId = that.dataForm.id || resp.id
                    
                    // 2. 再保存角色权限
                    const permissions = that.rightPermissions.map(item => item.id)
                    that.$http('/role/save-permissions', 'POST', {
                        roleId: roleId,
                        permissions: permissions
                    }, true, function(resp2) {
                        that.loading = false
                        if (resp2.code === 0 || resp2.code === 200) {
                            that.$message.success('操作成功')
                            that.visible = false
                            that.$emit('refresh')
                        } else {
                            that.$message.error(resp2.msg || '权限保存失败')
                        }
                    })
                } else {
                    that.loading = false
                    that.$message.error(resp.msg || '操作失败')
                }
            })
        }
    })
}
```

```java
public R list(@RequestBody Map<String, Object> params) {
    // 分页处理
    int page = Integer.parseInt(params.get("page").toString());
    int length = Integer.parseInt(params.get("length").toString());
    int start = (page - 1) * length;

    ArrayList<HashMap> roleList = misRoleService.searchRoleListWithCount(params);
```
